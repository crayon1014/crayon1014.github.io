jquery动画效果

1.上下联动

			$(document).ready(function(){
				$(".tab>li").mouseenter(function(){
					$(this).addClass("active").siblings().removeClass("active");
					var idx=$(this).index();
					$(".main").eq(idx).addClass("selected").siblings().removeClass("selected");
				})
			})

2.轮播

	$(document).ready(function(){
		var idx=0;
		$(".wrap>ol>li").mouseenter(function(){
			idx+=1
			$(this).addClass("active").siblings().removeClass("active");
			$(".wrap>ul>li").eq($(this).index()).css({"left":650,"z-index":idx}).animate({"left":0},300);
		})
	})

3.手风琴

			$(function() {
				$(".wrap>ul>li").each(function(i, ele) {
					$(this).css("background-image", "url(img/" + (i + 1) + ".jpg)");
				}).mouseenter(function() {
					$(this).stop().animate({
						"width": 600
					}, 300).siblings().stop().animate({
						"width": 100
					}, 300)
				}).mouseleave(function() {
					$("li").stop().animate({
						"width": 200
					}, 300)
				})
			})

4.天猫导航

			$(function() {
				$(window).scroll(function() {
					var tp = $(document).scrollTop()
					if(tp >= 1000) {
						$("#bar").show(300)
					} else {
						$("#bar").hide(300)
					}

				})
				$("#bar li").click(function() {
					$(this).css("background", "#FFA500").siblings().css("background", "greenyellow")
					$(document).scrollTop($(this).index() * 500 + 1000)
				})
			})
